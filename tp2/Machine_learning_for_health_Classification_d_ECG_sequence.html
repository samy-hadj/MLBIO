# Fonction pour calculer plusieurs caractristiques statistiques  partir d'un tableau
def calculate_stats_features(x):
    """
    Cette fonction prend en entre un tableau numrique `x` et calcule plusieurs statistiques :
    - Valeur maximale
    - Valeur minimale
    - Moyenne
    - Mdiane
    - Variance

    Les calculs ignorent les valeurs NaN.

    Paramtre :
    x (array-like) : Tableau de donnes numriques.

    Retour :
    list : Une liste contenant les statistiques dans l'ordre suivant :
           [max_val, min_val, mean_val, median_val, variance_val].
    """

    # Calcul de la valeur maximale en ignorant les NaN
    max_val = np.nanmax(x)

    # Calcul de la valeur minimale en ignorant les NaN
    min_val = np.nanmin(x)

    # Calcul de la moyenne en ignorant les NaN
    mean_val = np.nanmean(x)

    # Calcul de la mdiane en ignorant les NaN
    median_val = np.nanmedian(x)

    # Calcul de la variance en ignorant les NaN
    variance_val = np.nanvar(x)

    # Retourner les statistiques sous forme de liste
    return [max_val, min_val, mean_val, median_val, variance_val]

Create a function named calculate_zero_crossing(x) that calculates the Zero Crossing of a signal x.

The zero crossing is defined as the number of times the signal changes sign. For this, you can use the signbit, diff, and nonzero functions from numpy.

# Fonction pour calculer le nombre de passages par zro dans un tableau
def calculate_zero_crossing(x):
    """
    Cette fonction calcule le nombre de passages par zro dans un tableau de donnes `x`.
    Un passage par zro se produit lorsque le signal change de signe (de positif  ngatif ou inversement).

    Paramtre :
    x (array-like) : Tableau de donnes numriques.

    Retour :
    int : Le nombre de passages par zro dtects dans le tableau.
    """

    # Dtermination du signe de chaque lment : True pour ngatif, False pour positif
    # np.signbit renvoie True si l'lment est ngatif, False sinon
    signs = np.signbit(x)

    # Identification des indices o le signe change
    # np.diff calcule la diffrence entre deux lments conscutifs
    # .nonzero()[0] retourne les indices o la diffrence n'est pas zro (changement de signe)
    zero_crossings = np.diff(signs).nonzero()[0]

    # Retourner le nombre de passages par zro dtects
    return len(zero_crossings)

# Fonction pour calculer la valeur quadratique moyenne (RMS - Root Mean Square) d'un tableau
def calculate_rms(x):
    """
    Cette fonction calcule la valeur quadratique moyenne (RMS) d'un tableau `x`.
    La RMS est une mesure statistique souvent utilise pour analyser des signaux
    ou des donnes numriques.

    Formule : RMS = sqrt(mean(x^2)), o les NaN sont ignors.

    Paramtre :
    x (array-like) : Tableau de donnes numriques.

    Retour :
    float : La valeur quadratique moyenne calcule.
    """

    # Calcul de la moyenne des carrs des lments en ignorant les NaN
    mean_of_squares = np.nanmean(x**2)

    # Calcul de la racine carre de la moyenne des carrs
    rms = np.sqrt(mean_of_squares)

    # Retourner la valeur quadratique moyenne
    return rms

# Fonction pour calculer l'entropie de Shannon d'un tableau
from scipy.stats import entropy  # Importation de la fonction entropy pour le calcul

def calculate_entropy(x):
    """
    Cette fonction calcule l'entropie de Shannon d'un tableau `x`.
    L'entropie mesure le niveau d'incertitude ou de diversit dans une distribution de probabilit.

    tapes principales :
    1. Construction de l'histogramme pour approximer la distribution de probabilit.
    2. Ajustement des frquences pour viter les problmes lis aux zros (logarithme de 0).
    3. Calcul de l'entropie avec base 2 (bits).

    Paramtre :
    x (array-like) : Tableau de donnes numriques.

    Retour :
    float : L'entropie de Shannon calcule.
    """

    # Construction de l'histogramme normalis
    # bins=50 divise les donnes en 50 intervalles gaux
    # density=True garantit que la somme des frquences dans l'histogramme est gale  1
    hist, bin_edges = np.histogram(x, bins=50, density=True)

    # Ajout d'une petite valeur pour viter log(0) lors du calcul de l'entropie
    hist += 1e-10  # vite les erreurs dues aux frquences nulles

    # Calcul de l'entropie de Shannon avec base 2
    shannon_entropy = entropy(hist, base=2)

    # Retourner l'entropie calcule
    return shannon_entropy

# Fonction pour extraire plusieurs caractristiques d'un tableau de donnes
def get_features(x):
    """
    Cette fonction extrait plusieurs caractristiques d'un tableau `x`,
    y compris des statistiques simples, des informations sur les passages par zro,
    l'nergie RMS, et l'entropie de Shannon.

    Caractristiques calcules :
    - Statistiques de base : Maximum, Minimum, Moyenne, Mdiane, Variance
    - Nombre de passages par zro
    - RMS (Root Mean Square)
    - Entropie de Shannon

    Paramtre :
    x (array-like) : Tableau de donnes numriques.

    Retour :
    dict : Un dictionnaire contenant toutes les caractristiques calcules.
    """

    # Calcul des caractristiques statistiques
    # Cette tape extrait les valeurs max, min, moyenne, mdiane et variance
    stats_features = calculate_stats_features(x)
    max_val, min_val, mean_val, median_val, variance_val = stats_features

    # Calcul du nombre de passages par zro
    # Un passage par zro correspond  un changement de signe dans les donnes
    zero_crossings = calculate_zero_crossing(x)

    # Calcul de la valeur RMS (Root Mean Square)
    # La RMS est une mesure d'nergie moyenne du signal
    rms = calculate_rms(x)

    # Calcul de l'entropie de Shannon
    # L'entropie mesure le niveau de dsordre ou de diversit dans les donnes
    shannon_entropy = calculate_entropy(x)

    # Combinaison de toutes les caractristiques dans un dictionnaire
    # Chaque cl correspond au nom de la caractristique, et la valeur est le rsultat calcul
    features = {
        "Max": max_val,              # Valeur maximale du tableau
        "Min": min_val,              # Valeur minimale du tableau
        "Mean": mean_val,            # Moyenne des valeurs
        "Median": median_val,        # Mdiane des valeurs
        "Variance": variance_val,    # Variance des valeurs
        "Zero_Crossings": zero_crossings,  # Nombre de passages par zro
        "RMS": rms,                  # Valeur quadratique moyenne
        "Shannon_Entropy": shannon_entropy  # Entropie de Shannon
    }

    # Retour du dictionnaire contenant toutes les caractristiques
    return features

def extract_features_from_signals(signals, labels):
    """
    Cette fonction extrait les caractristiques des signaux ECG  l'aide de la fonction `get_features`
    applique  chaque signal individuel.

    Paramtres :
    signals (DataFrame) : Les signaux ECG sous forme de DataFrame.
    labels (Series) : Les tiquettes correspondantes pour chaque signal.

    Retour :
    DataFrame : Un DataFrame contenant toutes les caractristiques extraites ainsi que les labels.

# Importation des bibliothques ncessaires
import numpy as np  # Bibliothque pour manipulation de matrices et de tableaux

# Fonction pour extraire les coefficients de Fourier dun signal ECG
def get_fourier_coefficients(ecg, n_coefficients=10):
    """
    Cette fonction calcule les coefficients de Fourier pour un signal ECG.

    Paramtres :
    ecg (ndarray) : Le signal ECG sous forme de tableau numpy.
    n_coefficients (int) : Le nombre de coefficients de Fourier  extraire. Par dfaut  10.

    Retour :
    ndarray : Les n_coefficients premiers coefficients de Fourier sous forme de tableau numpy.
    """

    # Calcul de la transforme de Fourier du signal ECG
    fourier_transform = np.fft.fft(ecg)  # Utilisation de la fonction fft pour la transforme de Fourier

    # Calcul des magnitudes des coefficients de Fourier
    magnitudes = np.abs(fourier_transform)  # Calcul des amplitudes des frquences

    # Slection des premiers n_coefficients coefficients de Fourier
    coefficients = magnitudes[:n_coefficients]  # Rcupration des amplitudes des premires frquences

    return coefficients  # Retourne les n_coefficients premiers coefficients de Fourier

import pywt  # Importation de la bibliothque PyWavelets
import numpy as np  # Importation de la bibliothque NumPy

# Dfinition de la fonction pour extraire les coefficients de la transforme en ondelettes (Wavelet)
def get_wavelet_coefficients(ecg, wavelet='db4', level=4):
    """
    Cette fonction extrait les coefficients de la transforme en ondelettes d'un signal ECG.

    Parameters:
    - ecg : array-like
        Signal ECG sous forme d'array de donnes.
    - wavelet : str, optional (default='db4')
        Type de la fonction d'ondelette utilise. Par dfaut, 'db4' (Daubechies 4).
    - level : int, optional (default=4)
        Niveau de dcomposition (niveau de profondeur de la transforme en ondelettes).

    Returns:
    - coeffs_flat : array
        Coefficients de la transforme en ondelettes sous forme aplatie.
    """
    # Effectue la dcomposition en ondelettes du signal ECG
    coeffs = pywt.wavedec(ecg, wavelet, level=level)

    # Aplatir la liste de coefficients (pour obtenir un array avec tous les coefficients)
    coeffs_flat = np.hstack(coeffs)

    return coeffs_flat

# Convertir les labels en entiers (si ncessaire)
le = LabelEncoder()
y_labels = le.fit_transform(labels)  # Encodage des labels en entiers

# Diviser les donnes en ensembles d'entranement et de test (40% de donnes pour le test)
X_train, X_test, y_train, y_test = train_test_split(X_signals, y_labels, test_size=0.4, random_state=42)

# Reshaper les donnes pour correspondre  l'entre d'un CNN 1D
X_train = X_train.values.reshape(X_train.shape[0], X_train.shape[1], 1)  # Shape adapt pour un CNN : (chantillons, timesteps, 1)
X_test = X_test.values.reshape(X_test.shape[0], X_test.shape[1], 1)    # Mme format pour les donnes de test

# Dfinition du modle CNN
model = Sequential()

# Premire couche de convolution 1D avec filtres de 32, taille de filtre 5 et activation ReLU
model.add(Conv1D(filters=32, kernel_size=5, activation='relu', input_shape=(X_train.shape[1], 1)))

# Couche de pooling 1D pour rduire les dimensions des donnes
model.add(MaxPooling1D(pool_size=2))

# Deuxime couche de convolution 1D avec filtres de 64 et taille de filtre 5
model.add(Conv1D(filters=64, kernel_size=5, activation='relu'))

# Deuxime couche de pooling 1D
model.add(MaxPooling1D(pool_size=2))

# Troisime couche de convolution 1D avec filtres de 128 et taille de filtre 5
model.add(Conv1D(filters=128, kernel_size=5, activation='relu'))

# Troisime couche de pooling 1D
model.add(MaxPooling1D(pool_size=2))

# Flattening de la sortie des couches convolutionnelles pour les connecter  la couche dense
model.add(Flatten())

# Couche dense avec 128 units et activation ReLU
model.add(Dense(128, activation='relu'))

# Dropout pour viter l'overfitting (perte de connectivit alatoire)
model.add(Dropout(0.5))

# Couche de sortie avec activation softmax pour la classification multiclass
model.add(Dense(len(np.unique(y_labels)), activation='softmax'))

# Compilation du modle avec l'optimiseur Adam et la fonction de perte sparse_categorical_crossentropy
model.compile(optimizer=Adam(), loss='sparse_categorical_crossentropy', metrics=['accuracy'])

# Entraner le modle avec les donnes d'entranement
history = model.fit(X_train, y_train, epochs=10, batch_size=64, validation_data=(X_test, y_test))

# valuer le modle sur le jeu de test
