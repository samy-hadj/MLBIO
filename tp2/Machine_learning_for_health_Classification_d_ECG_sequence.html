# Fonction pour calculer plusieurs caractéristiques statistiques à partir d'un tableau
def calculate_stats_features(x):
    """
    Cette fonction prend en entrée un tableau numérique `x` et calcule plusieurs statistiques :
    - Valeur maximale
    - Valeur minimale
    - Moyenne
    - Médiane
    - Variance

    Les calculs ignorent les valeurs NaN.

    Paramètre :
    x (array-like) : Tableau de données numériques.

    Retour :
    list : Une liste contenant les statistiques dans l'ordre suivant :
           [max_val, min_val, mean_val, median_val, variance_val].
    """

    # Calcul de la valeur maximale en ignorant les NaN
    max_val = np.nanmax(x)

    # Calcul de la valeur minimale en ignorant les NaN
    min_val = np.nanmin(x)

    # Calcul de la moyenne en ignorant les NaN
    mean_val = np.nanmean(x)

    # Calcul de la médiane en ignorant les NaN
    median_val = np.nanmedian(x)

    # Calcul de la variance en ignorant les NaN
    variance_val = np.nanvar(x)

    # Retourner les statistiques sous forme de liste
    return [max_val, min_val, mean_val, median_val, variance_val]

Create a function named calculate_zero_crossing(x) that calculates the Zero Crossing of a signal x.

The zero crossing is defined as the number of times the signal changes sign. For this, you can use the signbit, diff, and nonzero functions from numpy.

# Fonction pour calculer le nombre de passages par zéro dans un tableau
def calculate_zero_crossing(x):
    """
    Cette fonction calcule le nombre de passages par zéro dans un tableau de données `x`.
    Un passage par zéro se produit lorsque le signal change de signe (de positif à négatif ou inversement).

    Paramètre :
    x (array-like) : Tableau de données numériques.

    Retour :
    int : Le nombre de passages par zéro détectés dans le tableau.
    """

    # Détermination du signe de chaque élément : True pour négatif, False pour positif
    # np.signbit renvoie True si l'élément est négatif, False sinon
    signs = np.signbit(x)

    # Identification des indices où le signe change
    # np.diff calcule la différence entre deux éléments consécutifs
    # .nonzero()[0] retourne les indices où la différence n'est pas zéro (changement de signe)
    zero_crossings = np.diff(signs).nonzero()[0]

    # Retourner le nombre de passages par zéro détectés
    return len(zero_crossings)

# Fonction pour calculer la valeur quadratique moyenne (RMS - Root Mean Square) d'un tableau
def calculate_rms(x):
    """
    Cette fonction calcule la valeur quadratique moyenne (RMS) d'un tableau `x`.
    La RMS est une mesure statistique souvent utilisée pour analyser des signaux
    ou des données numériques.

    Formule : RMS = sqrt(mean(x^2)), où les NaN sont ignorés.

    Paramètre :
    x (array-like) : Tableau de données numériques.

    Retour :
    float : La valeur quadratique moyenne calculée.
    """

    # Calcul de la moyenne des carrés des éléments en ignorant les NaN
    mean_of_squares = np.nanmean(x**2)

    # Calcul de la racine carrée de la moyenne des carrés
    rms = np.sqrt(mean_of_squares)

    # Retourner la valeur quadratique moyenne
    return rms

# Fonction pour calculer l'entropie de Shannon d'un tableau
from scipy.stats import entropy  # Importation de la fonction entropy pour le calcul

def calculate_entropy(x):
    """
    Cette fonction calcule l'entropie de Shannon d'un tableau `x`.
    L'entropie mesure le niveau d'incertitude ou de diversité dans une distribution de probabilité.

    Étapes principales :
    1. Construction de l'histogramme pour approximer la distribution de probabilité.
    2. Ajustement des fréquences pour éviter les problèmes liés aux zéros (logarithme de 0).
    3. Calcul de l'entropie avec base 2 (bits).

    Paramètre :
    x (array-like) : Tableau de données numériques.

    Retour :
    float : L'entropie de Shannon calculée.
    """

    # Construction de l'histogramme normalisé
    # bins=50 divise les données en 50 intervalles égaux
    # density=True garantit que la somme des fréquences dans l'histogramme est égale à 1
    hist, bin_edges = np.histogram(x, bins=50, density=True)

    # Ajout d'une petite valeur pour éviter log(0) lors du calcul de l'entropie
    hist += 1e-10  # Évite les erreurs dues aux fréquences nulles

    # Calcul de l'entropie de Shannon avec base 2
    shannon_entropy = entropy(hist, base=2)

    # Retourner l'entropie calculée
    return shannon_entropy

# Fonction pour extraire plusieurs caractéristiques d'un tableau de données
def get_features(x):
    """
    Cette fonction extrait plusieurs caractéristiques d'un tableau `x`,
    y compris des statistiques simples, des informations sur les passages par zéro,
    l'énergie RMS, et l'entropie de Shannon.

    Caractéristiques calculées :
    - Statistiques de base : Maximum, Minimum, Moyenne, Médiane, Variance
    - Nombre de passages par zéro
    - RMS (Root Mean Square)
    - Entropie de Shannon

    Paramètre :
    x (array-like) : Tableau de données numériques.

    Retour :
    dict : Un dictionnaire contenant toutes les caractéristiques calculées.
    """

    # Calcul des caractéristiques statistiques
    # Cette étape extrait les valeurs max, min, moyenne, médiane et variance
    stats_features = calculate_stats_features(x)
    max_val, min_val, mean_val, median_val, variance_val = stats_features

    # Calcul du nombre de passages par zéro
    # Un passage par zéro correspond à un changement de signe dans les données
    zero_crossings = calculate_zero_crossing(x)

    # Calcul de la valeur RMS (Root Mean Square)
    # La RMS est une mesure d'énergie moyenne du signal
    rms = calculate_rms(x)

    # Calcul de l'entropie de Shannon
    # L'entropie mesure le niveau de désordre ou de diversité dans les données
    shannon_entropy = calculate_entropy(x)

    # Combinaison de toutes les caractéristiques dans un dictionnaire
    # Chaque clé correspond au nom de la caractéristique, et la valeur est le résultat calculé
    features = {
        "Max": max_val,              # Valeur maximale du tableau
        "Min": min_val,              # Valeur minimale du tableau
        "Mean": mean_val,            # Moyenne des valeurs
        "Median": median_val,        # Médiane des valeurs
        "Variance": variance_val,    # Variance des valeurs
        "Zero_Crossings": zero_crossings,  # Nombre de passages par zéro
        "RMS": rms,                  # Valeur quadratique moyenne
        "Shannon_Entropy": shannon_entropy  # Entropie de Shannon
    }

    # Retour du dictionnaire contenant toutes les caractéristiques
    return features

# Fonction pour extraire les caractéristiques des signaux ECG
def extract_features_from_signals(signals, labels):
    """
    Cette fonction extrait les caractéristiques des signaux ECG à l'aide de la fonction `get_features`
    appliquée à chaque signal individuel.

    Paramètres :
    signals (DataFrame) : Les signaux ECG sous forme de DataFrame.
    labels (Series) : Les étiquettes correspondantes pour chaque signal.

    Retour :
    DataFrame : Un DataFrame contenant toutes les caractéristiques extraites ainsi que les labels.
    """
    feature_list = []  # Liste pour stocker les caractéristiques extraites

    # Parcourir chaque signal et extraire les caractéristiques
    for i in range(len(signals)):
        signal = np.array(signals.iloc[i])  # Conversion du signal en tableau numpy
        features = get_features(signal)  # Extraction des caractéristiques à partir du signal
        features['Label'] = labels.iloc[i]  # Ajout du label correspondant au signal
        feature_list.append(features)  # Ajouter les caractéristiques au list

    return pd.DataFrame(feature_list)  # Convertir la liste en DataFrame

import numpy as np  # Bibliothèque pour manipulation de matrices et de tableaux

# Fonction pour extraire les coefficients de Fourier d’un signal ECG
def get_fourier_coefficients(ecg, n_coefficients=10):
    """
    Cette fonction calcule les coefficients de Fourier pour un signal ECG.

    Paramètres :
    ecg (ndarray) : Le signal ECG sous forme de tableau numpy.
    n_coefficients (int) : Le nombre de coefficients de Fourier à extraire. Par défaut à 10.

    Retour :
    ndarray : Les n_coefficients premiers coefficients de Fourier sous forme de tableau numpy.
    """

    # Calcul de la transformée de Fourier du signal ECG
    fourier_transform = np.fft.fft(ecg)  # Utilisation de la fonction fft pour la transformée de Fourier

    # Calcul des magnitudes des coefficients de Fourier
    magnitudes = np.abs(fourier_transform)  # Calcul des amplitudes des fréquences

    # Sélection des premiers n_coefficients coefficients de Fourier
    coefficients = magnitudes[:n_coefficients]  # Récupération des amplitudes des premières fréquences

    return coefficients  # Retourne les n_coefficients premiers coefficients de Fourier

import pywt  # Importation de la bibliothèque PyWavelets
import numpy as np  # Importation de la bibliothèque NumPy

# Définition de la fonction pour extraire les coefficients de la transformée en ondelettes (Wavelet)
def get_wavelet_coefficients(ecg, wavelet='db4', level=4):
    """
    Cette fonction extrait les coefficients de la transformée en ondelettes d'un signal ECG.

    Parameters:
    - ecg : array-like
        Signal ECG sous forme d'array de données.
    - wavelet : str, optional (default='db4')
        Type de la fonction d'ondelette utilisée. Par défaut, 'db4' (Daubechies 4).
    - level : int, optional (default=4)
        Niveau de décomposition (niveau de profondeur de la transformée en ondelettes).

    Returns:
    - coeffs_flat : array
        Coefficients de la transformée en ondelettes sous forme aplatie.
    """
    # Effectue la décomposition en ondelettes du signal ECG
    coeffs = pywt.wavedec(ecg, wavelet, level=level)

    # Aplatir la liste de coefficients (pour obtenir un array avec tous les coefficients)
    coeffs_flat = np.hstack(coeffs)

    return coeffs_flat

    • Transformée de Fourier : pour capturer les fréquences dominantes des signaux.
    • Transformée Wavelet : pour obtenir des caractéristiques à la fois temporelles et fréquentielles.
  • J’ai combiné ces différentes caractéristiques pour enrichir les données.

  • J’ai utilisé un Random Forest Classifier, un modèle robuste adapté pour traiter des ensembles complexes de données.
  • J’ai ajusté plusieurs hyperparamètres, tels que le nombre d’arbres, la profondeur maximale et le critère de division, afin d’améliorer la performance.

3. Évaluation du Modèle :¶

  • Les données ont été scindées en ensembles d’entraînement et de test (typiquement selon des ratios comme 80/20 ou 60/40, en fonction des expérimentations).

# Convertir les labels en entiers (si nécessaire)
le = LabelEncoder()
y_labels = le.fit_transform(labels)  # Encodage des labels en entiers

# Diviser les données en ensembles d'entraînement et de test (40% de données pour le test)
X_train, X_test, y_train, y_test = train_test_split(X_signals, y_labels, test_size=0.4, random_state=42)

# Reshaper les données pour correspondre à l'entrée d'un CNN 1D
X_train = X_train.values.reshape(X_train.shape[0], X_train.shape[1], 1)  # Shape adapté pour un CNN : (échantillons, timesteps, 1)
X_test = X_test.values.reshape(X_test.shape[0], X_test.shape[1], 1)    # Même format pour les données de test

# Définition du modèle CNN
model = Sequential()

# Première couche de convolution 1D avec filtres de 32, taille de filtre 5 et activation ReLU
model.add(Conv1D(filters=32, kernel_size=5, activation='relu', input_shape=(X_train.shape[1], 1)))

# Couche de pooling 1D pour réduire les dimensions des données
model.add(MaxPooling1D(pool_size=2))

# Deuxième couche de convolution 1D avec filtres de 64 et taille de filtre 5
model.add(Conv1D(filters=64, kernel_size=5, activation='relu'))

# Deuxième couche de pooling 1D
model.add(MaxPooling1D(pool_size=2))

# Troisième couche de convolution 1D avec filtres de 128 et taille de filtre 5
model.add(Conv1D(filters=128, kernel_size=5, activation='relu'))

# Troisième couche de pooling 1D
model.add(MaxPooling1D(pool_size=2))

# Flattening de la sortie des couches convolutionnelles pour les connecter à la couche dense
model.add(Flatten())

# Couche dense avec 128 unités et activation ReLU
model.add(Dense(128, activation='relu'))

# Dropout pour éviter l'overfitting (perte de connectivité aléatoire)
model.add(Dropout(0.5))

# Couche de sortie avec activation softmax pour la classification multiclass
model.add(Dense(len(np.unique(y_labels)), activation='softmax'))

# Compilation du modèle avec l'optimiseur Adam et la fonction de perte sparse_categorical_crossentropy
model.compile(optimizer=Adam(), loss='sparse_categorical_crossentropy', metrics=['accuracy'])

# Entraîner le modèle avec les données d'entraînement
history = model.fit(X_train, y_train, epochs=10, batch_size=64, validation_data=(X_test, y_test))

# Évaluer le modèle sur le jeu de test
test_loss, test_accuracy = model.evaluate(X_test, y_test)

print(f"Test Loss: {test_loss}")
